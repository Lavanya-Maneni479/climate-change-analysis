 #CHART 1: Global Temperature Trend
 #Yearly Avg - Calucalation
import numpy as np

# Calculate yearly averages
yearly_temp = temp_global.groupby('Year')['LandAverageTemperature'].mean().reset_index()
yearly_temp.columns = ['Year', 'Avg_Temp']

print(f"Years: {yearly_temp['Year'].min()} to {yearly_temp['Year'].max()}")
print(f"Rows: {len(yearly_temp)}")

# Create chart
fig, ax = plt.subplots(figsize=(14, 6))

# Plot main line
ax.plot(yearly_temp['Year'],
        yearly_temp['Avg_Temp'],
        color='#e74c3c',
        linewidth=2.5,
        marker='o',
        markersize=3,
        label='Avg Temperature')

# Shading under line
ax.fill_between(yearly_temp['Year'],
                yearly_temp['Avg_Temp'],
                alpha=0.15,
                color='#e74c3c')

# Trend line
z = np.polyfit(yearly_temp['Year'], yearly_temp['Avg_Temp'], 1)
p = np.poly1d(z)
ax.plot(yearly_temp['Year'],
        p(yearly_temp['Year']),
        color='#c0392b',
        linewidth=1.5,
        linestyle='--',
        label='Trend line',
        alpha=0.7)

# Labels
ax.set_title('Global Average Temperature Trend (1900-2021)',
             fontsize=15, fontweight='bold', pad=15)
ax.set_xlabel('Year', fontsize=12)
ax.set_ylabel('Temperature (C)', fontsize=12)
ax.legend(fontsize=10)

# Annotation for hottest year
hottest = yearly_temp.loc[yearly_temp['Avg_Temp'].idxmax()]
ax.annotate(
    f"Hottest: {int(hottest['Year'])} ({hottest['Avg_Temp']:.2f}C)",
    xy=(hottest['Year'], hottest['Avg_Temp']),
    xytext=(hottest['Year']-20, hottest['Avg_Temp']+0.1),
    arrowprops=dict(arrowstyle='->', color='black'),
    fontsize=10, color='#c0392b', fontweight='bold'
)

plt.tight_layout()
ax.set_ylim(7.5, 10.5)  # zoom Y axis to show changes clearly

# Save
save_path = 'C:/Users/DELL/Downloads/Climate_Change_Analysis/'
plt.savefig(save_path + 'chart1_temperature_trend.png',
            dpi=150, bbox_inches='tight')
plt.show()
